.logon bhtest/dbc,dbc;

.set width 255
.set maxerror 25
.set errorout stdout


CREATE DATABASE D1 FROM Sysdba   
	AS PERMANENT = 0, SPOOL = 0;

CREATE PROFILE "p1" AS
SPOOL=1024
TEMPORARY=NULL
ACCOUNT=NULL
DEFAULT DATABASE=NULL
PASSWORD = (EXPIRE=NULL,MINCHAR=NULL,MAXCHAR=NULL,MAXLOGONATTEMPTS=NULL,LOCKEDUSEREXPIRE=NULL,REUSE=NULL,DIGITS=NULL,RESTRICTWORDS=NULL,SPECCHAR=NULL);

CREATE USER "u1" FROM "D1"
AS PERM = 0
PASSWORD = u1
STARTUP = ''
FALLBACK
NO BEFORE JOURNAL
NO AFTER JOURNAL
COLLATION = HOST
DEFAULT CHARACTER SET LATIN
DATEFORM=ANSIDATE
PROFILE="P1"
DEFAULT ROLE=ALL;


Create Role R1;
grant TABLE, SELECT, INSERT, UPDATE, DELETE on D1 to R1;

grant R1 to u1;

.logon bhtest/u1,u1;
database D1;


create table t21 ( sno int, sal int);

insert into t21 values(1,2);
.quit
